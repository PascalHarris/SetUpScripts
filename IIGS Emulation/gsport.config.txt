# --- Networking (persisted F4 settings; keys are the raw variable names) ---
# AppleTalk bridging to a Netatalk AppleShare server (see netatalk-server-setup.md).
g_appletalk_bridging = 1
# Network number the bridge uses. Must match the server's seeded net.
#   Server as a seed router with '-net 1'  -> set this to 1 (recommended).
#   Server in the startup range (no router) -> the bridge usually can't connect.
g_appletalk_network_hint = 1
# Host interface index for the bridge/Uthernet. VERIFY against the "Interface
# List:" shown in F4 -> Ethernet Card Configuration, then uncomment/set it:
# g_ethernet_interface = 0
# Uthernet card in slot 3 for TCP/IP (BBS via Marinetti). 0 = off, 1 = on.
# Runs simultaneously with AppleTalk (slot 3 vs slot 1 - no conflict).
# On the IIgs, install Marinetti + the Uthernet *I* (CS8900A) link layer, NOT
# Uthernet II. See marinetti-bbs-setup.md.
g_ethernet = 1
# Set to 1 to show AppleTalk diagnostics while debugging the bridge.
g_appletalk_diagnostics = 0

# RAM size, ZipGS acceleration and ROM selection are set from the F4 menu and
# persist here after Save. First-boot checklist:
#   1. ROM File Selection          -> /opt/gsport/ROM   (ROM03; required for AppleTalk)
#   2. Memory                       -> keep <=8MB while stabilising (14MB patch can
#                                      trip a non-canonical ROM); raise once stable
#   3. ZipGS / Speed                -> 8.0MHz (Zip) for stable accel; Unlimited can
#                                      break timing-sensitive software
#   4. Ethernet Card Configuration  -> set "Use Interface Number" to your eth0
#   5. Serial Port Configuration    -> Port 1 (slot 2) = "Only use socket 6502"
#
# Serial (slots 1/2), Uthernet (slot 3) and AppleTalk run simultaneously.
# On ROM03, AppleTalk is enabled on the IIgs via Control Panel -> Slots ->
# Slot 1 = AppleTalk (NOT slot 7). AppleTalk takes the printer port (slot 1);
# use the modem port (slot 2 / socket 6502) for the RS232 bridge.
